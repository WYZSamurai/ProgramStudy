### 二维数组特定列排序

    a.sort_unstable_by(|x, y| x[0].cmp(&y[0]));// x[0]<=y[0]

### 判断质数
    fn is_prime(n: &u32) -> bool {
        if *n <= 1 {
            return false;
        }
        if *n % 2 == 0 && *n != 2 {
            return false;
        }
        for i in (3..=(*n as f32).sqrt() as u32).step_by(2) {
            if *n % i == 0 {
                return false;
            }
        }
        true
    }
