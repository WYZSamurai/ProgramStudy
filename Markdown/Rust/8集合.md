### Vector
#### 迭代器
    let mut v = vec![1, 2, 3, 4, 5];
    for i in v.iter_mut() {
        *i += 1;
        println!("{}", i);
    }
#### 在vector1中删除所有vector2的元素
使用 **retain()** 保留vector2的补集

    let mut vector1 = vec![1, 2, 3, 4, 5];
    let vector2 = vec![2, 4];
    vector1.retain(|x| !vector2.contains(x));
    println!("{:?}", vector1); // prints [1, 3, 5]
